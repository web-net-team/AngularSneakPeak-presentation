Property And Event Bindings

Property Bindings

Event Bindings
<todo-cmp [model]="todo" (complete)="onCompletingTodo(todo)"></todo-cmp>

Property And Event Code Behind

@Component({
  selector: 'todo-cmp',
  properties: ['model'],
  events: ['complete']
})
// Component controller
class TodoCmp {
	model;
	complete = new EventEmitter();
	
	onCompletedButton() {
		this.complete.next(); // this fires an event
	}  
}
